            const employeeCostEl = document.getElementById('employeeCost');

            const aiCostEl = document.getElementById('aiCost');

            const monthlySavingsEl = document.getElementById('monthlySavings');

            const yearlySavingsEl = document.getElementById('yearlySavings');


            const aiAgentCost = 30000;

            aiCostEl.textContent = `${aiAgentCost.toLocaleString('ru-RU')} руб.`;

            

            const ctx = document.getElementById('savingsChart').getContext('2d');

            let savingsChart = new Chart(ctx, {

                type: 'doughnut',

                data: {

                    labels: ['Расходы на сотрудников', 'Стоимость ИИ-агента'],

                    datasets: [{

                        data: [180000, aiAgentCost],

                        backgroundColor: ['#ef4444', '#22c55e'],

                        borderColor: '#ffffff',

                        borderWidth: 4,

                    }]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    cutout: '70%',

                    plugins: {

                        legend: {

                            position: 'bottom',

                        },

                        tooltip: {

                            callbacks: {

                                label: function(context) {

                                    let label = context.label || '';

                                    if (label) {

                                        label += ': ';

                                    }

                                    if (context.parsed !== null) {

                                        label += new Intl.NumberFormat('ru-RU', { style: 'currency', currency: 'RUB' }).format(context.parsed);

                                    }

                                    return label;

                                }

                            }

                        }

                    }

                }

            });


            function updateCalculator() {

                const numEmployees = parseInt(employeesInput.value);

                const avgSalary = parseInt(salaryInput.value);

                

                employeesValue.textContent = numEmployees;

                salaryValue.textContent = avgSalary.toLocaleString('ru-RU');


                const totalEmployeeCost = numEmployees * avgSalary;

                const monthlySavings = totalEmployeeCost - aiAgentCost;

                const yearlySavings = monthlySavings * 12;


                employeeCostEl.textContent = `${totalEmployeeCost.toLocaleString('ru-RU')} руб.`;

                monthlySavingsEl.textContent = `${monthlySavings > 0 ? monthlySavings.toLocaleString('ru-RU') : 0} руб.`;

                yearlySavingsEl.textContent = `${yearlySavings > 0 ? yearlySavings.toLocaleString('ru-RU') : 0} руб.`;

                

                savingsChart.data.datasets[0].data[0] = totalEmployeeCost;

                savingsChart.update();

            }


            employeesInput.addEventListener('input', updateCalculator);

            salaryInput.addEventListener('input', updateCalculator);

            

            updateCalculator();

        });
